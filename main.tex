\begin{frame}{Mathematical Formulation}
    We optimize the parameters $\theta$ by minimizing the empirical risk. 
    
    \vspace{0.3cm}
    
    \begin{infobox}[\faCalculator \hspace{2pt} Objective Function]
        \begin{equation}
            \mathcal{L}(\theta) = \frac{1}{N} \sum_{i=1}^{N} \ell(f_\theta(\bm{x}_i), y_i) + \lambda \mathcal{R}(\theta)
        \end{equation}
        where:
        \begin{itemize}
            \item[\faAngleRight] $\ell(\cdot, \cdot)$ is the loss function (e.g., \alert{Cross-Entropy}).
            \item[\faAngleRight] $\mathcal{R}(\theta)$ is the regularization term to prevent overfitting.
            \item[\faAngleRight] $\lambda$ is the regularization hyperparameter.
        \end{itemize}
    \end{infobox}
\end{frame}
